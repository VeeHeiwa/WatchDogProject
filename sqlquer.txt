SQL QUERIES FOR MS SQL SERVER:
-----------------------------------------------------------------------------------------------------------------
*** CREATING TABLE ::
CREATE TABLE TABLENAME (
NAME VARCHAR (20),      // VARCHAR ALLOCATES SPACE ACORDING TO THE CHARACTERS ADDED.
ROLLNO INT,
);
-----------------------------------------------------------------------------------------------------------------

*** INSERTING INTO TABLE::
 INSERT INTO student
 VALUES ('Harry',02,'Thane',35);
     OR
 INSERT INTO table_name (column1,column2,column3,...)

*** Deleting From Table::
   DELETE FROM STUDENT WHERE Name = 'HARRY';

*** Altering table::
  // To add a column in a table, use the following syntax:
         ALTER TABLE table_name
         ADD column_name datatype
 // Deleting a column from a table
   ALTER TABLE table_name
   DROP COLUMN column_name

 // RENAMING EXISTING COLUMN
     SP_RENAME 'TABLENAME.OLD_COLUMN_NAME','NEW COLUMN_NAME','COLUMN'

  //CHANGING DATATYPE OF A COLUMN IN TABLE.

     ALTER TABLE table_name
     ALTER COLUMN column_name datatype

         UPDATE table_name
         SET column1=value1,column2=value2,...
         WHERE some_column=some_value;
-----------------------------------------------------------------------------------------------------------------
*** SQL AND/OR CLAUSE
      SELECT COLUMN1, COLUMN2...COLUMNN
      FROM TABLE_NAME
WHERE CONDITION-1 (AND|OR) CONDTION-2;
-----------------------------------------------------------------------------------------------------------------
*** sql between clause::
    select column1,column2...cloumnN
    from table_name
where column_name in (val-1, val-2......val-n);
-----------------------------------------------------------------------------------------------------------------
*** ORDER BY CLAUSE::
     SELECT COLUMN1,COLUMN2....COLUMNN
     FROM TABLE_NAME
     WHERE CONDITION
     ORDER BY COLUMN_NAME (ASC/DEC);

*** PROC TO INSERT INTO TABLE::
       EXAMPLE: 
 	  CREATE PROCEDURE STUDENT_proc
         @NAME VARCHAR (20),
	 @DOB VARCHAR (20),
	 @ROLL_NO INT,
	 @COURSE VARCHAR (20)
  AS 
     BEGIN
      INSERT INTO dbo.student
 	    (NAME,DOB,ROLL_NO,COURSE)
      VALUES
         	    (
		@NAME,
		@DOB,
		@ROLL_NO,
		@COURSE)
        END

  // EXECUTING THE PROCEDURE:
 exec STUDENT_proc 'sanket','14/05/1887',5,'comp'
-----------------------------------------------------------------------------------------------------------------
*** DELETING STORED PROCEDURE::
   EXAMPLE:
      CREATE PROCEDURE PROC_STUDENT_DELETE
         @NAME VARCHAR (20)
            AS
              BEGIN
            	 DELETE FROM STUDENT
        	   WHERE  NAME= @NAME;
    END

 EXEC PROC_STUDENT_DELETE 'SANKET DHAN'

*** UPDATING USING PROCEDURE::
   EXAMPLE:
      CREATE PROCEDURE STUDENT_UPDATE
                 @NAME VARCHAR (20),

                  @VALUE VARCHAR (20)
                    AS 
                     BEGIN
                         UPDATE STUDENT
                	 SET COURSE = @VALUE
	                 WHERE NAME = @NAME;
	       END
   EXEC STUDENT_UPDATE 'NINAD SAWANT','MEDICAL'

*** FUNCTIONS IN SQL::

  []    AVERAGE::
    SELECT AVG(COLUMN_NAME) FROM TABLENAME
  []    COUNTING NUMBER OF RECORDS IN TABLE::
    SELECT COUNT (*) FROM STUDENT
  []    COUNTING NUMBER OF VALUES OF A SPECIFIED COLUMN::
    SELECT COUNT(COLUMN_NAME) FROM TABLE_NAME
  []    COUNTING DISTINCT VALUES(DIFFERENT)
    SELECT COUNT (DISTINCT COLUMN_NAME) FROM TABLE_NAME
  []    DISPLAY FIRST//REQUIRED ELEMENTS FROM TABLE
    SELECT TOP 1 (COLUMN_NAME) FROM TABLE_NAME
  []    TO DISPLAY FIRST ROW FROM TABLE
    SELECT TOP 1 * FROM TABLE_NAME 
  []    TO DISPLAY LAST ROW FROM TABLE
    SELECT TOP 1  * FROM TABLE_NAME
    ORDER BY  COLUMN_NAME DESC
  []    GROUP BY FUNCTION
